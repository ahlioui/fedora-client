114     /**
115      * Creates a JMS manager using jndi properties to start a connection
116      * to a JMS provider. If jndi properties are null an attempt will
117      * be made to create a connection based on a <code>java:comp/env</code>
118      * context
119      *
120      * A connection must have a clientId in order to create durable
121      * subscriptions. This clientId can either be set administratively
122      * on the Connection object created in the JNDI store or by providing
123      * a non-null value for the clientId parameter of this method.
124      *
125      * @param jndiProps
126      * @param clientId
127      * @throws MessagingException
128      */
129     public JMSManager(Properties jndiProps, String clientId)
130             throws MessagingException {
131 
132         this.jndiProps = jndiProps;
133 
134         // Check properties
135         if (jndiProps == null) {
136             // Attempt a lookup to see if properties are provided by the container
137             String providerUrl = null;
138             try {
139                 providerUrl = (String) jndiLookup(Context.PROVIDER_URL);
140             } catch (MessagingException me) {
141                 providerUrl = null;
142             }
143             if(providerUrl == null) {
144                 throw new MessagingException("JMS connection properties must either be "
145                         + "provided by the container or by a non-null properties file "
146                         + "containing, at minimum, a value for "
147                         + Context.INITIAL_CONTEXT_FACTORY + " and "
148                         + Context.PROVIDER_URL + ".");
149             }
150         } else {
151             String icf = jndiProps.getProperty(Context.INITIAL_CONTEXT_FACTORY);
152             if (icf == null) {
153                 throw new MessagingException("A value for "
154                         + Context.INITIAL_CONTEXT_FACTORY
155                         + " must be included in the JNDI properties.");
156             }
157 
158             String pUrl = jndiProps.getProperty(Context.PROVIDER_URL);
159             if (pUrl == null) {
160                 throw new MessagingException("A value for "
161                         + Context.PROVIDER_URL
162                         + "must be included in the JNDI properties.");
163             }
164         }
165 
166         connectToJMS(clientId);
167     }

182     /**
183      * Creates a Destination if the Destination has not already been created.
184      *
185      * @param name - the name of the destination to create
186      * @param type - the destination type (topic or queue)
187      * @param fTransacted - determines whether the session will maintain transactions
188      * @param ackMode - determines the session acknowledgment mode
189      * @throws MessagingException
190      */
191     public Destination createDestination(String name,
192                                   DestinationType type,
193                                   boolean fTransacted,
194                                   int ackMode) throws MessagingException {
195         // If the destination already exists, just return it
196         JMSDestination jmsDest = jmsDestinations.get(name);
197         if (jmsDest != null) {
198             return jmsDest.destination;
199         }
200 
201         // Create the new destination and store it
202         Session session;
203         try {
204             session = connection.createSession(fTransacted, ackMode);
205         } catch (JMSException e) {
206             throw new MessagingException(e.getMessage(), e);
207         }
208 
209         // Look up the destination otherwise create it
210         Destination destination = null;
211         try {
212             destination = (Destination) jndiLookup(name);
213         } catch (MessagingException me) {
214             logger.debug("JNDI lookup for destination " + name + " failed. "
215                     + "Destination must be created.");
216             destination = null;
217         }
218         if (destination == null) {
219             // Create a topic or queue as specified
220             try {
221                 if (type.equals(DestinationType.Queue)) {
222                     logger.debug("setupDestination() - creating Queue" + name);
223                     destination = session.createQueue(name);
224                 } else {
225                     logger.debug("setupDestination() - creating Topic " + name);
226                     destination = session.createTopic(name);
227                 }
228             } catch (JMSException e) {
229                 throw new MessagingException(e.getMessage(), e);
230             }
231         }
232 
233         jmsDest = new JMSDestination(destination, session, null, null);
234 
235         jmsDestinations.put(name, jmsDest);
236 
237         return destination;
238     }

297         try {
298             Session s =

302             s.close();
303             return msg;

306         }

440     /**
441      * This is an asynchronous and durable listen. The caller provides a JMS
442      * callback interface reference, and any messages received for the given
443      * destination are provided through the onMessage() callback method. If the
444      * listener becomes unavailable the JMS provider will store messages
445      * received on the given topic until the listener reestablishes a connection
446      * and will then deliver those messages.
447      * Messages will be filtered based on the provided message selector.
448      *
449      * @param topic
450      *        the topic on which to listen
451      * @param messageSelector
452      *        selection criteria for filtering messages
453      * @param callback
454      *        the listener to call when a message is received
455      * @param subscriptionName
456      *        the name of the subscription
457      * @return The subscription name which can be used to stop, unsubscribe,
458      *         and reconnect to this topic
459      * @throws MessagingException
460      */
461     public String listenDurable(Topic topic,
462                                 String messageSelector,
463                                 MessageListener callback,
464                                 String subscriptionName)
465             throws MessagingException {
466         try {
467             String clientId = connection.getClientID();
468             if (clientId == null) {
469                 throw new MessagingException("A non-null client ID must be provided upon "
470                         + "creation of a JMSManager in order to create a JMS connection "
471                         + "capable of creating durable subscriptions.");
472             }
473             if (subscriptionName == null) {
474                 subscriptionName = topic.getTopicName();
475             }
476 
477             Session s =
478                     connection.createSession(false, Session.AUTO_ACKNOWLEDGE);
479             MessageConsumer c =
480                     s.createDurableSubscriber(topic, subscriptionName, messageSelector, false);
481             c.setMessageListener(callback);
482             durableSubscriptions.put(subscriptionName, c);
483         } catch (JMSException e) {
484             throw new MessagingException(e.getMessage(), e);
485         }
486 
487         if (logger.isDebugEnabled()) {
488             logger.debug("listen() - Asynchronous durable listen on topic " + topic);
489         }
490 
491         return subscriptionName;
492     }

565     /**
566      * Stops producers and consumers on a given destination.
567      * This has no effect on durable subscriptions.
568      *
569      * @param destName
570      * @throws MessagingException
571      */
572     public void stop(String destName) throws MessagingException {
573         try {
574             // Look for an existing destination for the given destination
575             //
576             JMSDestination jmsDest =
577                     jmsDestinations.get(destName);
578             if (jmsDest != null) {
579                 // Close out all JMS related state
580                 //
581                 if (jmsDest.producer != null) {
582                     jmsDest.producer.close();
583                     logger.debug("Closed producer for " + destName);
584                 }
585                 if (jmsDest.consumer != null) {
586                     jmsDest.consumer.close();
587                     logger.debug("Closed consumer for " + destName);
588                 }
589                 if (jmsDest.session != null) {
590                     jmsDest.session.close();
591                     logger.debug("Closed session for " + destName);
592                 }
593 
594                 jmsDest.destination = null;
595                 jmsDest.session = null;
596                 jmsDest.producer = null;
597                 jmsDest.consumer = null;
598 
599                 // Remove the JMS client entry
600                 //
601                 jmsDestinations.remove(destName);
602                 jmsDest = null;
603             }
604         } catch (JMSException e) {
605             throw new MessagingException(e.getMessage(), e);
606         }
607     }

609     /**
610      * Stops a durable message consumer. Note that this is not
611      * the same as unsubscribing. When a durable message consumer is
612      * restarted all messages received since it was stopped will be
613      * delivered.
614      *
615      * @param subscriptionName - the name of the subscription
616      * @throws MessagingException
617      */
618     public void stopDurable(String subscriptionName) throws MessagingException {
619         try {
620             MessageConsumer durableSubscriber =
621                     durableSubscriptions.get(subscriptionName);
622             if (durableSubscriber != null) {
623                 durableSubscriber.close();
624             }
625         } catch (JMSException jmse) {
626             throw new MessagingException("Exception encountered attempting to "
627                     + "stop durable subscription with name: "
628                     + subscriptionName + ". Exception message: "
629                     + jmse.getMessage(), jmse);
630         }
631     }

633     /**
634      * Removes the durable subscription with the given name.
635      *
636      * @param subscriptionName - name of the durable subscription
637      * @throws MessagingException
638      */
639     public void unsubscribeDurable(String subscriptionName)
640             throws MessagingException {
641         try {
642             Session session =
643                 connection.createSession(false, Session.AUTO_ACKNOWLEDGE);
644             MessageConsumer consumer = durableSubscriptions.get(subscriptionName);
645             if(consumer != null) {
646                 consumer.close();
647             }
648             session.unsubscribe(subscriptionName);
649         } catch (JMSException jmse) {
650             String errMsg = "Unable to unsubscribe from subscription with name: "
651                           + subscriptionName + " due to exception: "
652                           + jmse.getMessage();
653             logger.debug(errMsg, jmse);
654             throw new MessagingException(errMsg, jmse);
655         }
656     }
