168     @Test
169     public void testManagedChecksum() throws Exception {
170         FedoraResponse response = null;
171         String dsid = "testManagedChecksum";
172         File f = new File("src/test/resources/21.edit.essay.zip");
173         assertTrue(f.exists());
174 
175         String[] types = {"MD5", "SHA-1"};
176         String checksum;
177 
178         for (String type : types) {
179             dsid = dsid + type;
180             checksum = ChecksumUtility.checksum(type, new FileInputStream(f));
181             response = addDatastream(testPid, dsid).controlGroup("M").content(f)
182                     .checksumType(type).checksum(checksum).execute(fedora());
183             assertEquals(201, response.getStatus());
184             GetDatastreamResponse r = getDatastream(testPid, dsid).validateChecksum(true).execute(fedora());
185             assertTrue(r.isChecksumValid());
186         }
