37  public class GetDatastreamTest extends BaseFedoraRequestTest {
38  
39      @Test
40      public void testGetDatastreamAsHtml() throws Exception {
41          FedoraResponse response = null;
42  
43          response = getDatastream(testPid, "DC").format("html").execute(fedora());
44          assertEquals(200, response.getStatus());
45          String result = response.getEntity(String.class);
46  
47          assertXpathEvaluatesTo("Datastream Profile HTML Presentation",
48                                 "/html/head/title",
49                                 buildTestDocument(result));
50      }
51  
52      @Test
53      public void testGetDatastreamAsXml() throws Exception {
54          GetDatastreamResponse response = null;
55  
56          response = getDatastream(testPid, "DC").format("xml").execute(fedora());
57          assertEquals(200, response.getStatus());
58  
59          DatastreamProfile profile = response.getDatastreamProfile();
60          assertEquals(testPid, profile.getPid());
61          assertEquals("DC", profile.getDsID());
62      }
63  
64      private Document buildTestDocument(String html) throws Exception {
65          TolerantSaxDocumentBuilder tolerantSaxDocumentBuilder =
66                  new TolerantSaxDocumentBuilder(XMLUnit.newTestParser());
67          HTMLDocumentBuilder htmlDocumentBuilder =
68                  new HTMLDocumentBuilder(tolerantSaxDocumentBuilder);
69          return htmlDocumentBuilder.parse(html);
70      }
