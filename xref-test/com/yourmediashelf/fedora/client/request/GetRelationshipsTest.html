48      @Test
49      public void testGetRelationships() throws Exception {
50          FedoraResponse response = null;
51          String subject = String.format("info:fedora/%s", testPid);
52          String predicate = "urn:foo/testGetRelationships";
53          String object = "Able was I ere I saw Elba";
54  
55          // first add a relationship
56          response =
57                  fedora().execute(addRelationship(testPid).subject(subject)
58                          .predicate(predicate).object(object).isLiteral(true));
59          assertEquals(200, response.getStatus());
60  
61          // now get it
62          response =
63                  fedora().execute(getRelationships(testPid).subject(subject)
64                          .predicate(predicate));
65          assertEquals(200, response.getStatus());
66  
67          Model model = ModelFactory.createDefaultModel();
68          model.read(response.getEntityInputStream(), null, FileUtils.langXML);
69          StmtIterator it = model.listStatements();
70          Statement s;
71          while (it.hasNext()) {
72              s = it.next();
73              assertEquals(subject, s.getSubject().toString());
74              assertEquals(predicate, s.getPredicate().toString());
75              assertEquals(object, s.getObject().toString());
76          }
77      }

92  
93          // now get it
94          response =
95                  getRelationships(testPid).subject(subject)
96                          .predicate(predicate).execute(fedora());
97          assertEquals(200, response.getStatus());
98  
99          Model model = ModelFactory.createDefaultModel();
100         model.read(response.getEntityInputStream(), null, FileUtils.langXML);
101         StmtIterator it = model.listStatements();
102         Statement s;
103         while (it.hasNext()) {
104             s = it.next();
105             assertEquals(subject, s.getSubject().toString());
106             assertEquals(predicate, s.getPredicate().toString());
107             assertEquals(object, s.getLiteral().getLexicalForm());
108             assertEquals(datatype, s.getLiteral().getDatatypeURI());
109         }
