41          String dsId = "testListDatastreams";
42          // first, add an inline datastream
43          String content = "<foo>bar</foo>";
44          addDatastream(testPid, dsId).content(content).execute(fedora());
45  
46          // list datastreams
47          ListDatastreamsResponse response = listDatastreams(testPid).execute(fedora());
48          assertEquals(200, response.getStatus());
49          assertEquals(testPid, response.getPid());
50          // response baseUrl is appending a /
51          String baseUrl = getCredentials().getBaseUrl().toString() + "/";
52          assertEquals(baseUrl, response.getBaseUrl());
53          assertNull(response.getAsOfDateTime());
54  
55          List<DatastreamType> datastreams = response.getDatastreams();
56          boolean found = false;
57          for (DatastreamType d : datastreams) {
58              if (d.getDsid().equals(dsId)) {
59                  found = true;
60              }
61          }
62          assertTrue("Expected to find dsId among the list of datastreams", found);
63      }
