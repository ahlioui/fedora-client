38  public class PurgeRelationshipTest extends BaseFedoraRequestTest {
39  
40      @Test
41      public void testPurgeRelationship() throws Exception {
42          FedoraResponse response = null;
43          Model model = null;
44          Statement s = null;
45          String subject = String.format("info:fedora/%s", testPid);
46          String predicate = "urn:foo/p";
47          String object = "你好";
48  
49          // first add a relationship
50          response = fedora().execute(addRelationship(testPid).subject(subject).predicate(predicate).object(object).isLiteral(true));
51          assertEquals(200, response.getStatus());
52  
53          // verify it was added
54          response = fedora().execute(getRelationships(testPid).subject(subject).predicate(predicate));
55          assertEquals(200, response.getStatus());
56          model = ModelFactory.createDefaultModel();
57          model.read(response.getEntityInputStream(), null, FileUtils.langXML);
58          StmtIterator it = model.listStatements();
59          while (it.hasNext()) {
60              s = it.next();
61              assertEquals(subject, s.getSubject().toString());
62              assertEquals(predicate, s.getPredicate().toString());
63              assertEquals(object, s.getObject().toString());
64          }
65  
66          // now delete it
67          response = fedora().execute(purgeRelationship(testPid).subject(subject).predicate(predicate).object(object).isLiteral(true));
68          assertEquals(204, response.getStatus());
69  
70          // verify it's gone
71          response = fedora().execute(getRelationships(testPid).subject(subject).predicate(predicate));
72          assertEquals(200, response.getStatus());
73          model = ModelFactory.createDefaultModel();
74          model.read(response.getEntityInputStream(), null, FileUtils.langXML);
75          it = model.listStatements();
76          while (it.hasNext()) {
77              s = it.next();
78              assertFalse(predicate.equals(s.getPredicate().toString()));
79              assertFalse(object.equals(s.getObject().toString()));
80          }
81      }
