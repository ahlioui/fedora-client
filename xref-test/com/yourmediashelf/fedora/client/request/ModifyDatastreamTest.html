45      @Test
46      public void testModifyDatastreamContent() throws Exception {
47          String dsId = "testModifyDatastreamContent";
48          // first, add an inline datastream
49          String content = "<foo>bar</foo>";
50          FedoraResponse response;
51          response =
52                  addDatastream(testPid, dsId).content(content).execute(fedora());
53          assertEquals(201, response.getStatus());
54  
55          // verify datastream content before modify
56          response = getDatastreamDissemination(testPid, dsId).execute(fedora());
57          assertEquals(200, response.getStatus());
58          assertXMLEqual(content, response.getEntity(String.class));
59  
60          // verify datastream properties before modify
61          DatastreamProfileResponse dspResponse;
62          dspResponse =
63                  getDatastream(testPid, dsId).format("xml").execute(fedora());
64          assertEquals(200, response.getStatus());
65          DatastreamProfile profile = dspResponse.getDatastreamProfile();
66          assertEquals(testPid, profile.getPid());
67          assertEquals(dsId, profile.getDsID());
68          assertEquals("", profile.getDsLabel());
69          assertEquals(String.format("%s.0", dsId), profile.getDsVersionID());
70          assertNotNull(profile.getDsCreateDate());
71          assertEquals("A", profile.getDsState());
72          assertEquals("text/xml", profile.getDsMIME());
73          assertEquals("", profile.getDsFormatURI());
74          assertEquals("X", profile.getDsControlGroup());
75          assertEquals(16, profile.getDsSize().intValue());
76          assertEquals("true", profile.getDsVersionable());
77          assertEquals("", profile.getDsInfoType());
78          assertEquals(String.format("%s+%s+%s.0", testPid, dsId, dsId), profile
79                  .getDsLocation());
80          assertEquals("", profile.getDsLocationType());
81          assertEquals("DISABLED", profile.getDsChecksumType());
82          assertEquals("none", profile.getDsChecksum());
83  
84          // now modify the content (and label, while we're at it)
85          content = "<baz>quux</baz>";
86          String newDsLabel = "asdf";
87          response =
88                  modifyDatastream(testPid, dsId).content(content)
89                          .dsLabel(newDsLabel).execute(fedora());
90          assertEquals(200, response.getStatus());
91  
92          // verify datastream content after modify
93          response = getDatastreamDissemination(testPid, dsId).execute(fedora());
94          assertEquals(200, response.getStatus());
95          assertXMLEqual(content, response.getEntity(String.class));
96  
97          // verify datastream properties after modify
98          dspResponse =
99                  getDatastream(testPid, dsId).format("xml").execute(fedora());
100         assertEquals(200, dspResponse.getStatus());
101         profile = dspResponse.getDatastreamProfile();
102 
103         assertEquals(testPid, profile.getPid());
104         assertEquals(dsId, profile.getDsID());
105         assertEquals(newDsLabel, profile.getDsLabel());
106         assertEquals(String.format("%s.1", dsId), profile.getDsVersionID());
107         assertNotNull(profile.getDsCreateDate());
108         assertEquals("A", profile.getDsState());
109         assertEquals("text/xml", profile.getDsMIME());
110         assertEquals("", profile.getDsFormatURI());
111         assertEquals("X", profile.getDsControlGroup());
112         assertEquals(17, profile.getDsSize().intValue());
113         assertEquals("true", profile.getDsVersionable());
114         assertEquals("", profile.getDsInfoType());
115         assertEquals(String.format("%s+%s+%s.1", testPid, dsId, dsId), profile
116                 .getDsLocation());
117         assertEquals("", profile.getDsLocationType());
118         assertEquals("DISABLED", profile.getDsChecksumType());

136     @Test
137     public void testNullAndEmptyPidAndDsId() throws Exception {
138         String dsId = "testNullAndEmptyPidAndDsId";
139         try {
140             modifyDatastream(null, dsId).logMessage("test null pid").execute(fedora());
141             fail("Null pid should throw an IllegalArgumentException");
142         } catch (IllegalArgumentException expected) {}
143         try {
144             modifyDatastream("", dsId).logMessage("test empty pid").execute(fedora());
145             fail("Empty pid should throw an IllegalArgumentException");
146         } catch (IllegalArgumentException expected) {}
147         try {
148             modifyDatastream(null, null).logMessage("test null pid & dsid").execute(fedora());
149             fail("Null pid & dsId should throw an IllegalArgumentException");
150         } catch (IllegalArgumentException expected) {}
151         try {
152             modifyDatastream("", "").logMessage("test empty pid & dsid").execute(fedora());
153             fail("Empty pid & dsId should throw an IllegalArgumentException");
154         } catch (IllegalArgumentException expected) {}
155         try {
156             modifyDatastream(testPid, null).logMessage("test null dsid").execute(fedora());
157             fail("Null dsId should throw an IllegalArgumentException");
158         } catch (IllegalArgumentException expected) {}
159         try {
160             modifyDatastream(testPid, "").logMessage("test empty dsid").execute(fedora());
161             fail("Empty dsId should throw an IllegalArgumentException");
162         } catch (IllegalArgumentException expected) {}
163     }
