43      @Test
44      public void testModifyDatastreamContent() throws Exception {
45          String dsId = "testModifyDatastreamContent";
46          // first, add an inline datastream
47          String content = "<foo>bar</foo>";
48          FedoraResponse response;
49          response = addDatastream(testPid, dsId).content(content).execute(fedora());
50          assertEquals(201, response.getStatus());
51  
52          // verify datastream content before modify
53          response = getDatastreamDissemination(testPid, dsId).execute(fedora());
54          assertEquals(200, response.getStatus());
55          assertXMLEqual(content, response.getEntity(String.class));
56  
57          // verify datastream properties before modify
58  
59          DatastreamProfileResponse dspResponse;
60          dspResponse = getDatastream(testPid, dsId).format("xml").execute(fedora());
61          assertEquals(200, response.getStatus());
62          DatastreamProfile profile = dspResponse.getDatastreamProfile();
63          assertEquals(testPid, profile.getPid());
64          assertEquals(dsId, profile.getDsID());
65          assertEquals("", profile.getDsLabel());
66          assertEquals(String.format("%s.0", dsId), profile.getDsVersionID());
67          assertNotNull(profile.getDsCreateDate());
68          assertEquals("A", profile.getDsState());
69          assertEquals("text/xml", profile.getDsMIME());
70          assertEquals("", profile.getDsFormatURI());
71          assertEquals("X", profile.getDsControlGroup());
72          assertEquals(16, profile.getDsSize().intValue());
73          assertEquals("true", profile.getDsVersionable());
74          assertEquals("", profile.getDsInfoType());
75          assertEquals(String.format("%s+%s+%s.0", testPid, dsId, dsId), profile.getDsLocation());
76          assertEquals("", profile.getDsLocationType());
77          assertEquals("DISABLED", profile.getDsChecksumType());
78          assertEquals("none", profile.getDsChecksum());
79  
80          // now modify it
81          content = "<baz>quux</baz>";
82          String newDsLabel = "asdf";
83          response = modifyDatastream(testPid, dsId).content(content).dsLabel(newDsLabel).execute(fedora());
84          assertEquals(200, response.getStatus());
85  
86          // verify datastream content after modify
87          response = getDatastreamDissemination(testPid, dsId).execute(fedora());
88          assertEquals(200, response.getStatus());
89          assertXMLEqual(content, response.getEntity(String.class));
90  
91          // verify datastream properties after modify
92          dspResponse = getDatastream(testPid, dsId).format("xml").execute(fedora());
93          assertEquals(200, response.getStatus());
94          profile = dspResponse.getDatastreamProfile();
95  
96          assertEquals(testPid, profile.getPid());
97          assertEquals(dsId, profile.getDsID());
98          assertEquals(newDsLabel, profile.getDsLabel());
99          assertEquals(String.format("%s.1", dsId), profile.getDsVersionID());
100         assertNotNull(profile.getDsCreateDate());
101         assertEquals("A", profile.getDsState());
102         assertEquals("text/xml", profile.getDsMIME());
103         assertEquals("", profile.getDsFormatURI());
104         assertEquals("X", profile.getDsControlGroup());
105         assertEquals(17, profile.getDsSize().intValue());
106         assertEquals("true", profile.getDsVersionable());
107         assertEquals("", profile.getDsInfoType());
108         assertEquals(String.format("%s+%s+%s.1", testPid, dsId, dsId), profile.getDsLocation());
109         assertEquals("", profile.getDsLocationType());
110         assertEquals("DISABLED", profile.getDsChecksumType());
111         assertEquals("none", profile.getDsChecksum());
112     }
