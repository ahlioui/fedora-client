117     private void testCanonicalization(String prefix) throws Exception {
118         String input = String.format("%s-input.xml", prefix);
119         String canonical = String.format("%s-canonical.xml", prefix);
120         InputStream in = new FileInputStream(new File(testDir, input));
121         ByteArrayOutputStream bout = new ByteArrayOutputStream();
122         XmlSerializer.canonicalize(in, bout);
123         String control = readFileAsString(new File(testDir, canonical));
124         assertEquals(input + " did not match " + canonical, control, bout
125                 .toString("UTF-8"));
126     }
127 
128     private static String readFileAsString(File file) throws IOException {
129         byte[] buffer = new byte[(int) file.length()];
130         BufferedInputStream f =
131                 new BufferedInputStream(new FileInputStream(file));
132         f.read(buffer);
133         return new String(buffer, "UTF-8");
134     }
