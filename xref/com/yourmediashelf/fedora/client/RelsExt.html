22  import java.io.ByteArrayOutputStream;
23  import java.io.UnsupportedEncodingException;
24  
25  import com.hp.hpl.jena.rdf.model.Model;
26  import com.hp.hpl.jena.rdf.model.ModelFactory;
27  import com.hp.hpl.jena.rdf.model.Property;
28  import com.hp.hpl.jena.rdf.model.RDFNode;
29  import com.hp.hpl.jena.rdf.model.Resource;

31  /**
32   * <p>A builder for a Fedora 3.x RELS-EXT datastream.
33   * This class is only necessary for working with Fedora repositories that do not
34   * support the Relationships API (via the REST API), i.e., Fedora 3.3 and
35   * earlier.</p>
36   *
37   * <p>With Fedora 3.4.0 and later, it's recommended to use AddRelationship.</p>
38   *
39   * @author Edwin Shin
40   *
41   */
42  public class RelsExt {
43  	public static class Builder {
44  		private final String pid;
45  		private final Resource subject;
46  		private final Model model = ModelFactory.createDefaultModel();
47  
48  		public Builder(String pid) {
49  			if (pid.startsWith("info:fedora/")) {
50  				pid = pid.substring(12);
51  			}
52  			this.pid = pid;
53  			model.setNsPrefix("fedora-model", "info:fedora/fedora-system:def/model#");
54  			this.subject = model.createResource(String.format("info:fedora/%s", pid));
55  			addCModel("info:fedora/fedora-system:ContentModel-3.0");
56  		}
57  
58  		public RelsExt build() {
59  			RelsExt result = new RelsExt(pid, model);
60  			return result;
61  		}
62  
63  		public Builder addRelationship(String predicate, String object, boolean isLiteral, String datatype) {
64  		    Property p = model.createProperty(predicate);
65  		    RDFNode o;
66  		    if (isLiteral) {
67  		        o = model.createTypedLiteral(object, datatype);
68  		    } else {
69  		        o = model.createResource(object);
70  		    }
71  		    model.add(subject, p, o);
72  		    return this;
73  		}
74  
75  		public Builder addCModel(String cmodel) {
76  			Property p = model.createProperty("info:fedora/fedora-system:def/model#hasModel");
77  			if (!cmodel.startsWith("info:fedora/")) {
78  				cmodel = "info:fedora/" + cmodel;
79  			}
80  			RDFNode o = model.createResource(cmodel);
81  			model.add(subject, p, o);
82  			return this;
83  		}
84  	}
85  
86  	private final String pid;
87  	private final Model model;
88  
89  	private RelsExt(String pid, Model model) {
90  		this.pid = pid;
91  		this.model = model;
92  	}
93  
94  	@Override
95      public String toString() {
96  	    ByteArrayOutputStream bout = new ByteArrayOutputStream();
97  	    model.write(bout);
98  	    try {
99              return bout.toString("UTF-8");
100         } catch (UnsupportedEncodingException e) {
101             // TODO Auto-generated catch block
102             e.printStackTrace();
103             return null;
104         }
105 	}
106 
107 	public String getPid() {
108 		return pid;
109 	}
110 }
