190     @Override
191     public ModifyDatastreamResponse execute(FedoraClient fedora) throws FedoraClientException {
192         WebResource wr = fedora.resource();
193         String path = String.format("objects/%s/datastreams/%s", pid, dsId);
194         wr = wr.path(path).queryParams(getQueryParams());
195 
196         ClientResponse response = null;
197         if (content == null) {
198             response = wr.put(ClientResponse.class);
199         } else if (content instanceof String) {
200             response = wr.type(MediaType.TEXT_XML_TYPE).put(ClientResponse.class,
201                                                              content);
202         } else if (content instanceof File) {
203             File f = (File) content;
204             String mimeType = fedora.getMimeType(f);
205             response = wr.type(mimeType).put(ClientResponse.class, f);
206         } else {
207             throw new IllegalArgumentException("unknown request content type");
208         }
209         return new ModifyDatastreamResponse(response);
