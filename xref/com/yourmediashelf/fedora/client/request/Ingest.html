174     @Override
175     public IngestResponse execute(FedoraClient fedora) throws FedoraClientException {
176         ClientResponse response = null;
177         String path;
178         if (pid == null || pid.isEmpty()) {
179             path = "objects/new";
180         } else {
181             path = String.format("objects/%s", pid);
182         }
183 
184         WebResource wr = fedora.resource().path(path).queryParams(getQueryParams());
185 
186         if (content == null) {
187             response = wr.post(ClientResponse.class);
188         } else if (content instanceof String) {
189             response =
190                     wr.type(MediaType.TEXT_XML_TYPE).post(ClientResponse.class,
191                                                           content);
192         } else if (content instanceof File) {
193             File f = (File) content;
194             response =
195                     wr.type(MediaType.TEXT_XML_TYPE).post(ClientResponse.class,
196                                                           f);
197         } else {
198             throw new IllegalArgumentException("unknown request content type");
199         }
200         return new IngestResponse(response);
201     }
