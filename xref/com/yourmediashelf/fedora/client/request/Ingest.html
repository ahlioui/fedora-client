89      @Override
90      public IngestResponse execute(FedoraClient fedora) throws FedoraClientException {
91          ClientResponse response = null;
92          String path;
93          if (pid == null || pid.isEmpty()) {
94              path = "objects/new";
95          } else {
96              path = String.format("objects/%s", pid);
97          }
98  
99          WebResource wr = fedora.resource().path(path).queryParams(getQueryParams());
100 
101         if (content == null) {
102             response = wr.post(ClientResponse.class);
103         } else if (content instanceof String) {
104             response =
105                     wr.type(MediaType.TEXT_XML_TYPE).post(ClientResponse.class,
106                                                           content);
107         } else if (content instanceof File) {
108             File f = (File) content;
109             response =
110                     wr.type(MediaType.TEXT_XML_TYPE).post(ClientResponse.class,
111                                                           f);
112         } else {
113             throw new IllegalArgumentException("unknown request content type");
114         }
115         return new IngestResponse(response);
116     }
