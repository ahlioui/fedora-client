21  package com.yourmediashelf.fedora.client.request;

23  import java.util.List;
24  
25  import javax.ws.rs.core.MultivaluedMap;
26  
27  import com.sun.jersey.core.util.MultivaluedMapImpl;
28  import com.yourmediashelf.fedora.client.FedoraClient;
29  import com.yourmediashelf.fedora.client.FedoraClientException;
30  import com.yourmediashelf.fedora.client.response.FedoraResponse;

32  /**
33   * Base class for Fedora REST API requests.
34   *
35   * @author Edwin Shin
36   */
37  public abstract class FedoraRequest<T> {
38  
39      private final MultivaluedMap<String, String> queryParams =
40              new MultivaluedMapImpl();
41  
42      /**
43       * <p>Execute this request using the supplied FedoraClient instance.</p>
44       *
45       * @param fedora an instance of FedoraClient
46       * @return the response
47       * @throws FedoraClientException if the HTTP status code of the response is
48       * >= 400.
49       */
50      abstract public FedoraResponse execute(FedoraClient fedora)
51              throws FedoraClientException;
52  
53      /**
54       * <p>
55       * Add an arbitrary query parameter and value to the method.
56       * </p>
57       * <p>
58       * This method is intended as mild future-proofing, allowing fedora-client
59       * to be used with future or unreleased versions of Fedora that may support
60       * additional or different query parameters than fedora-client is aware of.
61       *
62       * @param key
63       *        the query parameter name
64       * @param value
65       *        the value
66       * @return this builder
67       */
68      @SuppressWarnings("unchecked")
69      public T xParam(String key, String value) {
70          addQueryParam(key, value);
71          return (T) this;
72      }
73  
74      protected void addQueryParam(String key, String value) {
75          if (value != null) {
76              queryParams.add(key, value);
77          }
78      }
79  
80      protected List<String> getQueryParam(String key) {
81          return queryParams.get(key);
82      }
83  
84      protected String getFirstQueryParam(String key) {
85          return queryParams.getFirst(key);
86      }
87  
88      protected MultivaluedMap<String, String> getQueryParams() {
89          return queryParams;
90      }
91  }
