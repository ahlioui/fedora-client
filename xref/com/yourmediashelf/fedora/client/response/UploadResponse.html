31  public class UploadResponse extends FedoraResponseImpl {
32      private String uploadLocation;

35          super(cr);
36      }
37  
38      /**
39       * Get the Fedora-generated upload url, typically for use in subsequent
40       * requests as the value of dsLocation.
41       *
42       * @return the upload url (in the form "uploaded://xxx")
43       * @throws FedoraClientException
44       */
45      public String getUploadLocation() throws FedoraClientException {
46          if (uploadLocation == null) {
47              uploadLocation = getEntity(String.class).trim();
48          }
49          return uploadLocation;
50      }
