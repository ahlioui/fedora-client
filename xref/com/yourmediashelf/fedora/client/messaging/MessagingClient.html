76      /**
77       * Creates a messaging client
78       *
79       * <h4>Client ID</h4>
80       * <p>
81       * The clientId provides applications with the opportunity to create
82       * multiple messaging clients and track them based on this identifier.
83       * The clientId is used within the MessagingClient when creating a
84       * connection for durable subscriptions.
85       * </p>
86       *
87       * <h4>Message Listener</h4>
88       * <p>
89       * A listener, the onMessage() method of which will be called when a
90       * message arrives from the messaging provider. See the documentation
91       * for javax.jms.MessageListener for more information.
92       * </p>
93       *
94       * <h4>Connection Properties</h4>
95       *
96       * <p>All of the following properties must be included:</p>
97       * <table border="1">
98       *   <th>Property</th>
99       *   <th>Description</th>
100      *   <th>Example Value</th>
101      *   <tr>
102      *     <td>java.naming.factory.initial (javax.naming.Context.INITIAL_CONTEXT_FACTORY)</td>
103      *     <td>The JNDI initial context factory which will allow lookup of the other attributes</td>
104      *     <td>org.apache.activemq.jndi.ActiveMQInitialContextFactory</td>
105      *   </tr>
106      *   <tr>
107      *     <td>java.naming.provider.url (javax.naming.Context.PROVIDER_URL)</td>
108      *     <td>The JNDI provider URL
109      *     <td>tcp://localhost:61616</td>
110      *   </tr>
111      *   <tr>
112      *     <td>connection.factory.name (org.fcrepo.server.messaging.JMSManager.CONNECTION_FACTORY)</td>
113      *     <td>The JNDI name of the connection factory needed to create a JMS Connection</td>
114      *     <td>ConnectionFactory</td>
115      *   </tr>
116      * </table>
117      * <p>One or more of the following properties must be specified:</p>
118      * <table border="1">
119      *   <th>Property Name</th>
120      *   <th>Description</th>
121      *   <th>Example Value</th>
122      *   <tr>
123      *     <td>topic.X (where X = name of subscription topic, example - topic.fedoraManagement)</td>
124      *     <td>A topic over which notification messages will be provided</td>
125      *     <td>fedora.apim.*</td>
126      *   </tr>
127      *   <tr>
128      *     <td>queue.X (where X = name of subscription queue, example - queue.fedoraManagement)</td>
129      *     <td>A queue through which notification messages will be provided</td>
130      *     <td>fedora.apim.update</td>
131      *   </tr>
132      * </table>
133      *
134      * <h4>Durable</h4>
135      * <p>
136      * Specifies whether the topics included in the connection properties should
137      * have durable consumers. If set to true, all topics listeners will be
138      * constructed as durable subscribers. If set to false, all topic listeners
139      * will be constructed as non-durable subscribers. This does not affect
140      * queue listeners.
141      * </p>
142      * <p>
143      * If there is a need for multiple topics, some of which are durable and some
144      * of which are not, then two MessagingClients should be created.
145      * One client would include topics needing durable subscribers and the other
146      * client would include topics not needing durable subscribers.
147      * A single MessageListener can be registered as the listener for both clients.
148      * </p>
149      *
150      * <h4>Message Selector</h4>
151      * <p>
152      * A JMS message selector allows a client to specify, by header field references
153      * and property references, the messages it is interested in. Only messages
154      * whose header and property values match the selector are delivered.
155      * See the javadoc for javax.jms.Message for more information about message selectors.
156      * </p>
157      *
158      * @param clientId identification value for this messaging client
159      * @param listener the listener which will be called when messages arrive
160      * @param connectionProperties set of properties necessary to connect to JMS provider
161      * @param messageSelector a selection which determines the messages to deliver
162      * @param durable determines if the underlying JMS subscribers are durable
163      * @throws MessagingException if listener is null or required properties are not set
164      */
165     public MessagingClient(String clientId,
166                               MessagingListener listener,
167                               Properties connectionProperties,
168                               String messageSelector,
169                               boolean durable)
170             throws MessagingException {
171 
172         // Check for a null listener
173         if (listener == null) {
174             throw new MessagingException("MessageListener may not be null");
175         }
176 
177         // Check for null properties
178         if (connectionProperties == null) {
179             throw new MessagingException("Connection properties may not be null");
180         }
181 
182         // Check for required property values
183         String initialContextFactory =
184             connectionProperties.getProperty(Context.INITIAL_CONTEXT_FACTORY);
185         String providerUrl =
186             connectionProperties.getProperty(Context.PROVIDER_URL);
187         String connectionFactoryName =
188             connectionProperties.getProperty(JMSManager.CONNECTION_FACTORY_NAME);
189 
190         if (initialContextFactory == null
191             || providerUrl == null
192             || connectionFactoryName == null) {
193             throw new MessagingException("Propery values for "
194                     + "'java.naming.factory.initial', "
195                     + "'java.naming.provider.url', and"
196                     + "'connection.factory.name' must be provided "
197                     + "in order to initialize a messaging client");
198         }
199 
200         // Check for valid client ID if durable subscriptions are required
201         if (durable) {
202             if (clientId == null || clientId.equals("")) {
203                 throw new MessagingException("ClientId must be "
204                         + "specified for durable subscriptions");
205             }
206         }
207 
208         m_clientId = clientId;
209         m_listener = listener;
210         m_connectionProperties = connectionProperties;
211         m_messageSelector = messageSelector;
212         m_durable = durable;
213     }

223     /**
224      * Starts the MessagingClient. This method must be called
225      * in order to receive messages.
226      *
227      * @param wait Set to true to wait until the startup process
228      *             is complete before returning. Set to false to
229      *             allow for asynchronous startup.
230      */
231     public void start(boolean wait) throws MessagingException {
232         Thread connector = new JMSBrokerConnector();
233         connector.start();
234 
235         if(wait) {
236             int maxWait = RETRY_INTERVAL * MAX_RETRIES;
237             int waitTime = 0;
238             while(!isConnected()) {
239                 if(waitTime < maxWait) {
240                     try {
241                         Thread.sleep(100);
242                         waitTime += 100;
243                     } catch (InterruptedException ie) {
244                         Thread.currentThread().interrupt();
245                     }
246                 } else {
247                     throw new MessagingException("Timeout reached waiting " +
248                                            "for messaging client to start.");
249                 }
250             }
251         }
252     }

258     /**
259      * Creates topics and queues and starts listeners
260      */
261     private void createDestinations() throws MessagingException {
262         try {
263             // Create Destinations based on properties
264             Enumeration<?> propertyNames = m_connectionProperties.keys();
265             while (propertyNames.hasMoreElements()) {
266                 String propertyName = (String) propertyNames.nextElement();
267                 if (propertyName.startsWith("topic.")) {
268                     String destinationName =
269                             m_connectionProperties.getProperty(propertyName);
270                     m_jmsManager.createDestination(destinationName,
271                                                    DestinationType.Topic);
272                 } else if (propertyName.startsWith("queue.")) {
273                     String destinationName =
274                             m_connectionProperties.getProperty(propertyName);
275                     m_jmsManager.createDestination(destinationName,
276                                                    DestinationType.Queue);
277                 }
278             }
279 
280             // Get destination list
281             List<Destination> destinations = m_jmsManager.getDestinations();
282 
283             // If there are no Destinations, throw an exception
284             if (destinations.size() == 0) {
285                 throw new MessagingException("No destinations available for "
286                         + "subscription, make sure that there is at least one topic "
287                         + "or queue specified in the connection properties.");
288             }
289 
290             // Subscribe
291             for (Destination destination : destinations) {
292                 if (m_durable && (destination instanceof Topic)) {
293                     m_jmsManager.listenDurable((Topic) destination,
294                                                m_messageSelector,
295                                                this,
296                                                null);
297                 } else {
298                     m_jmsManager.listen(destination, m_messageSelector, this);
299                 }
300             }
301         } catch (MessagingException me) {
302             logger.error("MessagingException encountered attempting to start "
303                     + "Messaging Client: " + m_clientId
304                     + ". Exception message: " + me.getMessage(), me);
305             throw me;
306         }
307     }

309     /**
310      * Stops the MessagingClient, shuts down connections. If the unsubscribe
311      * parameter is set to true, all durable subscriptions will be removed.
312      *
313      * @param unsubscribe
314      */
315     public void stop(boolean unsubscribe) throws MessagingException {
316         try {
317             if (unsubscribe) {
318                 m_jmsManager.unsubscribeAllDurable();
319             }
320             m_jmsManager.close();
321             m_jmsManager = null;
322             m_connected = false;
323         } catch (MessagingException me) {
324             logger.error("Messaging Exception encountered attempting to stop "
325                     + "Messaging Client: " + m_clientId
326                     + ". Exception message: " + me.getMessage(), me);
327             throw me;
328         }
329     }

341     /**
342      * Starts the connection to the JMS Broker. Retries on failure.
343      */
344     private class JMSBrokerConnector extends Thread {
345 
346         @Override
347         public void run() {
348             try {
349                 connect();
350                 createDestinations();
351                 m_connected = true;
352             } catch (MessagingException me) {
353                 throw new RuntimeException(me);
354             }
355         }
356 
357         private void connect() throws MessagingException {
358             int retries = 0;
359 
360             while(m_jmsManager == null && retries < MAX_RETRIES) {
361                 try {
362                     m_jmsManager = new JMSManager(m_connectionProperties, m_clientId);
363                 } catch(MessagingException me) {
364                     Throwable rootCause = me.getCause();
365                     while(rootCause.getCause() != null) {
366                         rootCause = rootCause.getCause();
367                     }
368 
369                     if(rootCause instanceof java.net.ConnectException) {
370                         try {
371                             sleep(RETRY_INTERVAL);
372                         } catch (InterruptedException ie) {
373                             Thread.currentThread().interrupt();
374                         }
375                         retries++;
376                     } else {
377                         throw me;
378                     }
379                 }
380             }
381 
382             if(m_jmsManager == null) {
383                 String errorMessage =
384                     "Unable to start JMS Messaging Client, " + MAX_RETRIES +
385                     " attempts were made, each attempt resulted in a " +
386                     "java.net.ConnectException. The messaging broker at " +
387                     m_connectionProperties.getProperty(Context.PROVIDER_URL) +
388                     " is not available";
389                 throw new RuntimeException(errorMessage);
390             }
391         }
