38  /**
39   * Representation of an API-M method call as an Atom entry.
40   * <ul>
41   * <li>atom:title corresponds to the method name, e.g. ingest</li>
42   * <li>Each atom:category corresponds to a method's argument:
43   * <ul>
44   * <li>The scheme indicates the argument name</li>
45   * <li>The term indicates the argument value. However, null values are
46   * indicated as "null", and non-null <code>xsd:base64Binary</code> values are
47   * indicated as "[OMITTED]".</li>
48   * <li>The label indicates the argument datatype</li>
49   * </ul>
50   * </li>
51   * <li>atom:content corresponds to the textual representation of the method's
52   * return value, noting the following:
53   * <ul>
54   * <li>Null values are represented as "null".</li>
55   * <li><code>fedora-types:ArrayOfString</code> values are represented as a
56   * comma-separated list, e.g. "value1, value2, value3".</li>
57   * <li>Non-null <code>xsd:base64Binary</code> values are not returned, and
58   * only indicated as "[OMITTED]".</li>
59   * <li>Non-null <code>fedora-types:Datastream</code> values are not returned,
60   * and only indicated as "[OMITTED]".</li>
61   * <li><code>fedora-types:RelationshipTuple</code> values are represented in
62   * Notation3 (N3).</li>
63   * </ul>
64   * </li>
65   * <li>atom:uri element of atom:author corresponds to the baseURL of the Fedora
66   * repository, e.g. http://localhost:8080/fedora.</li>
67   * <li>atom:summary corresponds to the PID of the method, if applicable.</li>
68   * </ul>
69   *
70   * @see <a href="http://atomenabled.org/developers/syndication/atom-format-spec.php">The Atom Syndication Format</a>
71   *
72   * @author Edwin Shin
73   */
74  public class AtomApimMessage {
75  
76      private static final Logger logger =
77              LoggerFactory.getLogger(AtomApimMessage.class);
78  
79      private final Abdera abdera = Abdera.getInstance();
80  
81      private final static String versionPredicate = "info:fedora/fedora-system:def/view#version";
82  
83      private final static String formatPredicate = "http://www.fedora.info/definitions/1/0/types/formatURI";
84  
85      private final String fedoraBaseUrl;
86  
87      private final String serverVersion;
88  
89      private final String format;
90  
91      private final String methodName;
92  
93      private final String pid;
94  
95      private final Date date;
96  
97      private final String author;
98  
99      private final String returnVal;
100 
101     private final Entry entry;
102 
103     public AtomApimMessage(String messageText) {
104         Parser parser = abdera.getParser();
105         Document<Entry> entryDoc = parser.parse(new StringReader(messageText));
106         entry = entryDoc.getRoot();
107         methodName = entry.getTitle();
108         date = entry.getUpdated();
109         author = entry.getAuthor().getName();
110         fedoraBaseUrl = entry.getAuthor().getUri().toString();
111 
112         pid = entry.getSummary();
113         returnVal = entry.getContent();
114 
115         serverVersion = getCategoryTerm(versionPredicate);
116         format = getCategoryTerm(formatPredicate);
117     }
118 
119     /**
120      * Serialization of the API-M message as an Atom entry. {@inheritDoc}
121      */
122     @Override
123     public String toString() {
124         Writer sWriter = new StringWriter();
125 
126         try {
127             entry.writeTo("prettyxml", sWriter);
128         } catch (IOException e) {
129             logger.error(e.getMessage());
130         }
131         return sWriter.toString();
132     }

190     /**
191      * Get the first atom:category term that matches the provided scheme.
192      *
193      * @param scheme
194      * @return the term or null if no match.
195      */
196     private String getCategoryTerm(String scheme) {
197         List<Category> categories = entry.getCategories(scheme);
198         if (categories.isEmpty()) {
199             return null;
200         } else {
201             return categories.get(0).getTerm();
202         }
203     }
