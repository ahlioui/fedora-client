68      /**
69       * Constructor for FedoraClient.
70       *
71       * @param fc credentials for a Fedora repository
72       */
73      public FedoraClient(FedoraCredentials fc) {
74          this.fc = fc;
75          nsCtx = new NamespaceContextImpl();
76          nsCtx.addNamespace("f", "info:fedora/fedora-system:def/foxml#");
77          mimeUtil = new MimeUtil2();
78          mimeUtil
79                  .registerMimeDetector("eu.medsea.mimeutil.detector.MagicMimeMimeDetector");
80  
81          // TODO validate fc
82          // null check for username & password
83          // null check for baseUrl
84  
85          // FIXME this isn't very security-minded
86          if (fc.getBaseUrl().toString().startsWith("https")) {
87              SSLContext ctx = null;
88              try {
89                  ctx = SSLContext.getInstance("SSL");
90                  ctx.init(null, null, null);
91              } catch (NoSuchAlgorithmException e) {
92                  logger.error(e.getMessage(), e);
93              } catch (KeyManagementException e) {
94                  logger.error(e.getMessage(), e);
95              }
96              ClientConfig config = new DefaultClientConfig();
97              // FIXME Consider using not-yet-commons-ssl hostnameverifier
98              // config by property: strict, none, etc.
99              config.getProperties()
100                     .put(HTTPSProperties.PROPERTY_HTTPS_PROPERTIES,
101                          new HTTPSProperties(null, ctx));
102             client = Client.create(config);
103         } else {
104             client = Client.create();
105         }
106         client.setFollowRedirects(true);
107     }

109     /**
110      * Convenience method for executing FedoraRequests. Note that this method
111      * returns a FedoraResponse and therefore requires an explicit cast if a
112      * specific implementation of FedoraResponse is desired.
113      *
114      * <p>The preferred usage is actually the reverse: pass an instance of
115      * FedoraClient to the FedoraRequest, e.g.</p>
116      *
117      * <pre>    IngestResponse r = new Ingest("my:pid").execute(client);</pre>
118      *
119      * <p>Finally, FedoraClient provides a number of static convenience methods
120      * for FedoraRequests. Therefore, with a static import, the following,
121      * abbreviated syntax can be used:</p>
122      *
123      * <pre>import static com.yourmediashelf.fedora.client.FedoraClient.*;
124      * ...
125      *     IngestResponse r = ingest("my:pid").execute(client);</pre>
126      *
127      * @param method a FedoraRequest
128      * @return the wrapped HTTP response
129      * @throws FedoraClientException
130      */
131     public FedoraResponse execute(FedoraRequest<?> method)
132             throws FedoraClientException {
133         return method.execute(this);
134     }

136     public WebResource resource() {
137         WebResource wr = client.resource(fc.getBaseUrl().toString());
138         wr
139                 .addFilter(new HTTPBasicAuthFilter(fc.getUsername(), fc
140                         .getPassword()));
141         //wr.addFilter(new LoggingFilter(System.out));
142         return wr;
143     }

150     public Date getLastModifiedDate(String pid) throws FedoraClientException {
151         FedoraResponse response = execute(getObjectXML(pid));
152 
153         String expr = "//f:objectProperties/f:property[@NAME='info:fedora/fedora-system:def/view#lastModifiedDate']/@VALUE";
154         String lastModifiedDate;
155         try {
156             lastModifiedDate = getXPath().evaluate(expr, new InputSource(response.getEntityInputStream()));
157         } catch (XPathExpressionException e) {
158             throw new FedoraClientException(e.getMessage(), e);
159         }
160         return DateUtility.parseXSDDateTime(lastModifiedDate).toDate();
161     }

163     public Date getLastModifiedDate(String pid, String dsId) throws FedoraClientException {
164         FedoraResponse response = execute(getDatastream(pid, dsId).format("xml"));
165         String expr = "/datastreamProfile/dsCreateDate";
166         String lastModifiedDate;
167         try {
168             lastModifiedDate = getXPath().evaluate(expr, new InputSource(response.getEntityInputStream()));
169         } catch (XPathExpressionException e) {
170             throw new FedoraClientException(e.getMessage(), e);
171         }
172         return DateUtility.parseXSDDateTime(lastModifiedDate).toDate();
173     }

175     private XPath getXPath() {
176         XPathFactory factory = XPathFactory.newInstance();
177         XPath xpath = factory.newXPath();
178         xpath.setNamespaceContext(nsCtx);
179         return xpath;
