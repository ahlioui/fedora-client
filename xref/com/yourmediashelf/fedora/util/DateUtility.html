19      /**
20       * <p>Regular expression string for the yearFrag:</p>
21       *
22       * <p>a numeral consisting of at least four decimal digits, optionally
23       * preceded by a minus sign; leading '0' digits are prohibited except to
24       * bring the digit count up to four.</p>
25       */
26      private final static String yearFrag = "(-?([1-9][0-9]{3,}|0[0-9]{3}))";
27  
28      /**
29       * <p>Regular expression string for the monthFrag:</p>
30       *
31       * <p>a numeral consisting of exactly two decimal digits.</p>
32       */
33      private final static String monthFrag = "(0[1-9]|1[0-2])";
34  
35      /**
36       * <p>Regular expression string for the dayFrag:</p>
37       *
38       * <p>a numeral consisting of exactly two decimal digits.</p>
39       *
40       * <p>Note that this regex does not enforce the day-of-month constraint,
41       * which states:</p>
42       *
43       * <p>The day value must be no more than 30 if month is one of 4, 6, 9,
44       * or 11; no more than 28 if month is 2 and year is not divisible 4,
45       * or is divisible by 100 but not by 400; and no more than 29 if month is
46       * 2 and year is divisible by 400, or by 4 but not by 100.</p>
47       */
48      private final static String dayFrag = "(0[1-9]|[12][0-9]|3[01])";
49  
50      /**
51       * <p>Regular expression string for the hourFrag:</p>
52       *
53       * <p>a numeral consisting of exactly two decimal digits.</p>
54       */
55      private final static String hourFrag = "([01][0-9]|2[0-3])";
56  
57      /**
58       * <p>Regular expression string for the minuteFrag:</p>
59       *
60       * <p>a numeral consisting of exactly two decimal digits.</p>
61       */
62      private final static String minuteFrag = "([0-5][0-9])";
63  
64      /**
65       * <p>Regular expression string for the secondFrag:</p>
66       *
67       * <p>a numeral consisting of exactly two decimal digits, or two decimal
68       * digits, a decimal point, and one or more trailing digits.</p>
69       */
70      private final static String secondFrag = "([0-5][0-9])(\\.([0-9]+))?";
71  
72      /**
73       * <p>Regular expression string for the endOfDayFrag:</p>
74       *
75       * <p>combines the {@link hourFrag}, {@link minuteFrag}, {@link minuteFrag},
76       * and their separators to represent midnight of the day, which is the first
77       * moment of the next day.</p>
78       */
79      private final static String endOfDayFrag = String.format("(%s:%s:%s|(24:00:00(\\.0+)?))", hourFrag, minuteFrag, secondFrag);
80  
81      /**
82       * <p>Regular expression string for the timezoneFrag: </p>
83       *
84       * <p>if present, specifies an offset between UTC and local time. Time zone
85       * offsets are a count of minutes (expressed in timezoneFrag  as a count of
86       * hours and minutes) that are added or subtracted from UTC time to get the
87       * "local" time.  'Z' is an alternative representation of the time zone
88       * offset '00:00', which is, of course, zero minutes from UTC.</p>
89       */
90      private final static String timezoneFrag = "(Z|(\\+|-)((0[0-9]|1[0-3]):[0-5][0-9]|14:00))?";
91  
92      /**
93       * Concatenation of the {@link yearFrag}, {@link monthFrag},
94       * {@link dayFrag}, {@link endOfDayFrag}, {@link timezoneFrag}, and their
95       * separators.
96       */
97      private final static String combined = String.format("%s-%s-%sT%s%s", yearFrag, monthFrag, dayFrag, endOfDayFrag, timezoneFrag);
98  
99      private final static Pattern XSD_DATETIME = Pattern.compile(combined);

102     /**
103      *
104      * @param input an XML Schema 1.1 dateTime
105      * @return a DateTime representing the input
106      * @see http://www.w3.org/TR/xmlschema11-2/#dateTime
107      */
108     public static DateTime parseXSDDateTime(String input) {
109         Matcher m = XSD_DATETIME.matcher(input);
110         if (!m.find()) {
111             throw new IllegalArgumentException(input + " is not a valid XML Schema 1.1 dateTime.");
112         }
113 
114         int year = Integer.parseInt(m.group(1));
115         int month = Integer.parseInt(m.group(3));
116         int day = Integer.parseInt(m.group(4));
117 
118         int hour = 0, minute = 0, second = 0, millis = 0;
119         boolean hasEndOfDayFrag = (m.group(11) != null);
120         if (!hasEndOfDayFrag) {
121             hour = Integer.parseInt(m.group(6));
122             minute = Integer.parseInt(m.group(7));
123             second = Integer.parseInt(m.group(8));
124             if (m.group(9) != null && !m.group(9).isEmpty()) {
125                 millis = (int)(Float.parseFloat(m.group(9)) * 1000);
126             }
127         }
128 
129         DateTimeZone zone = null;
130         if (m.group(13) != null) {
131             String tmp = m.group(13);
132             if (tmp.equals("Z")) {
133                 tmp = "+00:00";
134             }
135             zone = DateTimeZone.forID(tmp);
136         }
137         DateTime dt = new DateTime(year, month, day, hour, minute, second, millis, zone);
138 
139         if (hasEndOfDayFrag) {
140             return dt.plusDays(1);
141         }
142         return dt;
143     }

165     public static String getXSDDateTime(DateTime date) {
166         if (date.getMillis() == 0) {
167             return date.withZone(DateTimeZone.UTC).toString(ISODateTimeFormat.dateTimeNoMillis());
168         } else {
169             return date.withZone(DateTimeZone.UTC).toString(ISODateTimeFormat.dateTime());
