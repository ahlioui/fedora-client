21  package com.yourmediashelf.fedora.util;
22  
23  import java.io.BufferedInputStream;
24  import java.io.IOException;
25  import java.io.InputStream;
26  import java.security.MessageDigest;
27  import java.security.NoSuchAlgorithmException;
28  
29  /**
30   * @author Edwin Shin
31   */
32  public class ChecksumUtility {
33  
34      public static String checksum(String algorithm, String input)
35              throws Exception {
36          MessageDigest md = MessageDigest.getInstance(algorithm);
37          md.update(input.getBytes("UTF-8"));
38          return byteArray2HexString(md.digest());
39      }
40  
41      public static String checksum(String algorithm, InputStream is)
42              throws NoSuchAlgorithmException, IOException {
43          MessageDigest md = MessageDigest.getInstance(algorithm);
44          BufferedInputStream bis = new BufferedInputStream(is);
45          try {
46              byte[] buffer = new byte[1024];
47              int read;
48              do {
49                  read = bis.read(buffer);
50                  if (read > 0) md.update(buffer, 0, read);
51              } while (read != -1);
52          } finally {
53              is.close();
54          }
55          return byteArray2HexString(md.digest());
56      }
57  
58      public static String byteArray2HexString(byte[] buf) {
59          final char[] HEX_CHARS = "0123456789abcdef".toCharArray();
60  
61          char[] chars = new char[2 * buf.length];
62          for (int i = 0; i < buf.length; ++i) {
63              chars[2 * i] = HEX_CHARS[(buf[i] & 0xF0) >>> 4];
64              chars[2 * i + 1] = HEX_CHARS[buf[i] & 0x0F];
65          }
66          return new String(chars);
67      }
